1.3.📤 Upload Route (Core Pipeline Start)

// routes/upload.js
const express = require("express");
const multer = require("multer");
const crypto = require("crypto");
const { v4: uuidv4 } = require("uuid");
const fs = require("fs-extra");
const path = require("path");

const router = express.Router();

const storage = multer.diskStorage({
  destination: "server/storage/videos/",
  filename: (req, file, cb) => {
    cb(null, Date.now() + "-" + file.originalname);
  }
});

const upload = multer({ storage });

router.post("/", upload.single("video"), async (req, res) => {
  const filePath = req.file.path;

  // Generate SHA-256
  const fileBuffer = fs.readFileSync(filePath);
  const hash = crypto.createHash("sha256").update(fileBuffer).digest("hex");

  const incidentId = uuidv4();

  const incident = {
    id: incidentId,
    file: filePath,
    hash,
    metadata: req.body,
    status: "uploaded",
    createdAt: new Date(),
    detections: []
  };

  await fs.writeJson(`server/storage/${incidentId}.json`, incident);

  res.json({ message: "Uploaded", incidentId });
});

module.exports = router;

1.5.🔍 Verification Route
// routes/verify.js
const express = require("express");
const fs = require("fs-extra");

const router = express.Router();

// Get all incidents
router.get("/", async (req, res) => {
  const files = await fs.readdir("server/storage");
  const incidents = [];

  for (let f of files) {
    if (f.endsWith(".json")) {
      const data = await fs.readJson(`server/storage/${f}`);
      incidents.push(data);
    }
  }

  res.json(incidents);
});

// Approve plate
router.post("/approve", async (req, res) => {
  const { id, plate } = req.body;

  const file = `server/storage/${id}.json`;
  const incident = await fs.readJson(file);

  incident.status = "verified";
  incident.finalPlate = plate;
  incident.verifiedAt = new Date();

  await fs.writeJson(file, incident);

  res.json({ message: "Approved" });
});

module.exports = router;


1.6.📊 Dashboard Route
// routes/dashboard.js
const express = require("express");
const fs = require("fs-extra");

const router = express.Router();

router.get("/", async (req, res) => {
  const files = await fs.readdir("server/storage");

  let total = 0;
  let verified = 0;

  for (let f of files) {
    if (f.endsWith(".json")) {
      total++;
      const data = await fs.readJson(`server/storage/${f}`);
      if (data.status === "verified") verified++;
    }
  }

  res.json({
    total,
    verified,
    pending: total - verified
  });
});

module.exports = router;

2.4.🔍 verify.html
<h1>Verification</h1>
<div id="list"></div>

<script>
async function load() {
  const res = await fetch("http://localhost:3000/verify");
  const data = await res.json();

  const container = document.getElementById("list");

  data.forEach(item => {
    const div = document.createElement("div");

    div.innerHTML = `
      <p>ID: ${item.id}</p>
      <p>Status: ${item.status}</p>
      <input placeholder="Plate" id="plate-${item.id}" />
      <button onclick="approve('${item.id}')">Approve</button>
      <hr/>
    `;

    container.appendChild(div);
  });
}

async function approve(id) {
  const plate = document.getElementById(`plate-${id}`).value;

  await fetch("http://localhost:3000/verify/approve", {
    method: "POST",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({ id, plate })
  });

  alert("Approved");
  location.reload();
}

load();
</script>
